  <action>
    Create a standalone profiling script that times each step of the preprocessing pipeline
    independently. The script MUST:

    1. Accept a path to an audio file as sys.argv[1] (WebM, OGG, or WAV)
    2. Load and read the file as raw bytes (simulating what FastAPI receives)
    3. Time each step with time.perf_counter() — measure wall-clock time, not CPU time
    4. Print results in a readable table with milliseconds

    Structure (mirror the inference.py pipeline exactly, step by step):

    ```python
    import sys, time, io
    from pathlib import Path

    # Setup: read raw bytes (same as FastAPI UploadFile.read())
    audio_path = Path(sys.argv[1])
    audio_bytes = audio_path.read_bytes()
    content_type = "audio/webm" if audio_path.suffix == ".webm" else "audio/wav"

    print(f"Input: {audio_path.name}  ({len(audio_bytes)/1024:.1f} KB)")
    print("-" * 55)

    timings = {}

    # Step 1: ffmpeg conversion (pydub)
    t0 = time.perf_counter()
    from pydub import AudioSegment
    if "wav" not in content_type:
        fmt = "webm" if "webm" in content_type else "ogg"
        segment = AudioSegment.from_file(io.BytesIO(audio_bytes), format=fmt)
        buf = io.BytesIO(); segment.export(buf, format="wav"); wav_bytes = buf.getvalue()
    else:
        wav_bytes = audio_bytes
    timings["1. pydub+ffmpeg (WebM→WAV)"] = time.perf_counter() - t0

    # Step 2: librosa.load (WAV decode + resample to 22050 Hz)
    import librosa, numpy as np
    SAMPLE_RATE, DURATION = 22050, 3
    TARGET_LEN = SAMPLE_RATE * DURATION
    t0 = time.perf_counter()
    y, sr = librosa.load(io.BytesIO(wav_bytes), sr=SAMPLE_RATE, mono=True)
    timings["2. librosa.load (decode+resample)"] = time.perf_counter() - t0

    # Pad/truncate (negligible, not timed separately)
    y = np.pad(y, (0, max(0, TARGET_LEN - len(y))))[:TARGET_LEN]

    # Step 3: mel spectrogram + power_to_db
    t0 = time.perf_counter()
    mel = librosa.feature.melspectrogram(y=y, sr=SAMPLE_RATE, n_mels=128, n_fft=2048, hop_length=512)
    mel_db = librosa.power_to_db(mel, ref=np.max)
    timings["3. mel spectrogram + power_to_db"] = time.perf_counter() - t0

    # Step 4: scipy bilinear resize
    import scipy.ndimage
    t0 = time.perf_counter()
    h, w = mel_db.shape
    mel_r = scipy.ndimage.zoom(mel_db, (128/h, 128/w), order=1)
    timings["4. scipy resize (128×128)"] = time.perf_counter() - t0

    # Step 5: normalize + reshape
    t0 = time.perf_counter()
    tensor = mel_r[np.newaxis, np.newaxis, :, :].astype(np.float32) / 80.0
    timings["5. normalize + reshape"] = time.perf_counter() - t0

    # Step 6: onnxruntime inference
    import onnxruntime as ort
    MODEL = Path(__file__).parent.parent / "model" / "emotion_model.onnx"
    session = ort.InferenceSession(str(MODEL), providers=["CPUExecutionProvider"])
    input_name = session.get_inputs()[0].name
    # Warm up (first call loads weights into memory)
    _ = session.run(None, {input_name: tensor})
    t0 = time.perf_counter()
    logits = session.run(None, {input_name: tensor})[0]
    timings["6. onnxruntime session.run()"] = time.perf_counter() - t0

    # Print results
    total = sum(timings.values())
    print(f"{'Step':<40} {'ms':>8}  {'%':>6}")
    print("-" * 55)
    for step, elapsed in timings.items():
        ms = elapsed * 1000
        pct = elapsed / total * 100 if total > 0 else 0
        print(f"{step:<40} {ms:>8.1f}  {pct:>5.1f}%")
    print("-" * 55)
    print(f"{'TOTAL':<40} {total*1000:>8.1f}  100.0%")
    print()
    dominant = max(timings, key=timings.get)
    print(f"Dominant step: {dominant} ({timings[dominant]*1000:.1f} ms)")
    ```

    Avoid: importing from inference.py (keep profiler self-contained).
    Avoid: testing multiple runs in a loop — one realistic run is sufficient for diagnosis.
